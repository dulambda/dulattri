#!/bin/bash

# Parse arguments
while [[ "$#" -gt 0 ]]; do
    case "$1" in
        --input_folder) input_folder="$2"; shift 2;;
        --extension) extension="$2"; shift 2;;
        --backup_folder) backup_folder="$2"; shift 2;;
        --backup_archive_name) backup_archive_name="$2"; shift 2;;
        *) echo "Unknown parameter: $1"; exit 1;;
    esac
done

# Check required arguments
if [[ -z "$input_folder" || -z "$extension" || -z "$backup_folder" || -z "$backup_archive_name" ]]; then
    echo "Missing parameters"
    exit 1
fi

# Create backup folder
mkdir -p "$backup_folder"

# Find and copy files preserving folder structure
find "$input_folder" -type f -name "*.${extension}" | while read -r file; do
    rel_path=$(realpath --relative-to="$input_folder" "$file")
    target_dir="$backup_folder/$(dirname "$rel_path")"
    mkdir -p "$target_dir"
    cp "$file" "$target_dir/"
done

# Create archive
tar -czf "$backup_archive_name" "$backup_folder" &> /dev/null

# Done
echo "done"
